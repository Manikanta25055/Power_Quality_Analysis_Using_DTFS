\section{Theoretical Background}

\begin{frame}{Discrete Time Fourier Series (DTFS)}
    \begin{block}{Definition}
        DTFS decomposes periodic discrete-time signals into harmonic components.
    \end{block}
    
    \vspace{0.3cm}
    
    \textbf{Analysis Equation (Forward Transform):}
    \begin{equation}
        X[k] = \frac{1}{N} \sum_{n=0}^{N-1} x[n] \cdot e^{-j2\pi kn/N}
    \end{equation}
    
    \textbf{Synthesis Equation (Inverse Transform):}
    \begin{equation}
        x[n] = \sum_{k=0}^{N-1} X[k] \cdot e^{j2\pi kn/N}
    \end{equation}
    
    \vspace{0.2cm}
    
    where $k$ = harmonic index, $n$ = time sample index, $N$ = period length
\end{frame}

\begin{frame}{Power Quality Metrics}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Total Harmonic Distortion (THD):}
            \begin{equation}
                THD_F = \frac{\sqrt{\sum_{h=2}^{\infty} V_h^2}}{V_1} \times 100\%
            \end{equation}
            
            \textbf{Crest Factor (CF):}
            \begin{equation}
                CF = \frac{V_{peak}}{V_{RMS}}
            \end{equation}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{True Power Factor (TPF):}
            \begin{equation}
                TPF = \frac{1}{\sqrt{1 + (THD_F/100)^2}}
            \end{equation}
            
            \textbf{K-Factor:}
            \begin{equation}
                K = \frac{\sum_{h=1}^{h_{max}} h^2 \cdot I_h^2}{\sum_{h=1}^{h_{max}} I_h^2}
            \end{equation}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}[fragile]{Core Function 1: DTFS Analysis}
    \textbf{File:} \texttt{calculate\_dtfs.m}
    
    \vspace{0.2cm}
    
    \begin{lstlisting}[language=Matlab]
function [X_k, magnitude, phase, frequencies] = calculate_dtfs(x, fs, N)
    % Preallocate coefficient array
    X_k = zeros(N, 1);
    
    % Calculate DTFS coefficients
    for k = 0:(N-1)
        sum_value = 0;
        for n = 0:(N-1)
            complex_exp = exp(-1j*2*pi*k*n/N);
            sum_value = sum_value + x(n+1) * complex_exp;
        end
        X_k(k+1) = sum_value / N;
    end
    
    % Extract magnitude (scale for single-sided spectrum)
    magnitude = abs(X_k);
    magnitude(2:end) = 2 * magnitude(2:end);
    
    % Extract phase
    phase = angle(X_k);
    
    % Generate frequency axis
    frequencies = (0:(N-1))' * (fs/N);
end
    \end{lstlisting}
\end{frame}

\begin{frame}{DTFS Analysis - Key Points}
    \begin{block}{Direct Implementation}
        \begin{itemize}
            \item Direct implementation without FFT libraries
            \item Complexity: $O(N^2)$ for $N$ samples
            \item For $N = 200$: 40,000 complex multiplications
            \item Computation time: $\sim$0.05 seconds per analysis
        \end{itemize}
    \end{block}
    
    \vspace{0.3cm}
    
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Advantages:}
            \begin{itemize}
                \item Educational value
                \item Clear mathematical insight
                \item No toolbox dependency
                \item Exact frequency resolution
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Output Parameters:}
            \begin{itemize}
                \item Complex coefficients $X[k]$
                \item Magnitude spectrum
                \item Phase spectrum
                \item Frequency axis
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}[fragile]{Core Function 2: Inverse DTFS}
    \textbf{File:} \texttt{synthesize\_from\_dtfs.m}
    
    \vspace{0.2cm}
    
    \begin{lstlisting}[language=Matlab]
function [x_reconstructed, t] = synthesize_from_dtfs(X_k, fs, N, num_periods)
    total_samples = N * num_periods;
    x_reconstructed = zeros(total_samples, 1);
    t = (0:(total_samples-1))' / fs;
    
    % Inverse DTFS synthesis: x[n] = sum_k X[k]*exp(j*2*pi*k*n/N)
    for n = 0:(total_samples-1)
        sum_value = 0;
        for k = 0:(N-1)
            n_mod = mod(n, N);
            complex_exp = exp(1j * 2 * pi * k * n_mod / N);
            sum_value = sum_value + X_k(k+1) * complex_exp;
        end
        x_reconstructed(n+1) = sum_value;
    end
    
    x_reconstructed = real(x_reconstructed);
end
    \end{lstlisting}
    
    \vspace{0.2cm}
    
    \begin{block}{Purpose}
        Reconstructs time-domain signal from DTFS coefficients (essential for filtering).
    \end{block}
\end{frame}

function [signal, t, harmonic_info] = ...
         generate_distorted_signal(scenario, fs, f0, N, V_rms_fundamental)
    
    % Calculate fundamental peak voltage
    V_peak_fundamental = V_rms_fundamental * sqrt(2);
    
    % Generate time vector
    t = (0:N-1)' / fs;
    
    % Define harmonic content based on scenario
    switch scenario
        case 'led_lighting'
            harmonics = [1, 3, 5, 7, 9];
            percentages = [100, 18, 8, 4, 2];
            phases = [0, pi/6, -pi/4, pi/3, 0];
            expected_THD = 20.2;
            
        case 'motor_drive'
            harmonics = [1, 5, 7, 11, 13];
            percentages = [100, 20, 14, 9, 6];
            phases = [0, -pi/4, pi/4, -pi/6, pi/6];
            expected_THD = 26.7;
    end
    \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Core Function 3: Signal Generation (Part 2)}
    \begin{lstlisting}[language=Matlab]
    % Generate composite signal
    signal = zeros(N, 1);
    
    for i = 1:length(harmonics)
        h = harmonics(i);
        freq = h * f0;
        mag = V_peak_fundamental * (percentages(i) / 100);
        signal = signal + mag * sin(2*pi*freq*t + phases(i));
    end
    
    % Store harmonic information
    harmonic_info.harmonics = harmonics;
    harmonic_info.magnitudes = V_peak_fundamental*(percentages/100);

    harmonic_info.expected_THD = expected_THD;
end
    \end{lstlisting}

    \end{block}

\begin{frame}[fragile]{Core Function 4: Power Quality Metrics}

    \begin{lstlisting}[language=Matlab]
function metrics = power_quality_metrics(signal, magnitude, fs, f0)
    % Extract fundamental and harmonics
    V1_peak = magnitude(2);
    V1_rms = V1_peak / sqrt(2);
    harmonic_rms = magnitude(3:end) / sqrt(2);
    
    % Calculate THD-F
    harmonic_rms_total = sqrt(sum(harmonic_rms.^2));
    THD_F = (harmonic_rms_total/V1_rms)*100;
    
    % Calculate crest factor
    signal_rms = sqrt(mean(signal.^2));
    crest_factor = max(abs(signal)) / signal_rms;
    
    % Calculate power factor
    distortion_PF = 1 / sqrt(1 + (THD_F/100)^2);
    
    % Calculate K-factor
    K_factor = 1.0;
    for h = 2:length(harmonic_rms)
        K_factor = K_factor + h^2*(harmonic_rms(h-1)/V1_rms)^2;
    end
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Core Function 5: Harmonic Filtering}

function [X_k_filtered, filter_info] = ...
         design_harmonic_filter(X_k, N, filter_type, harmonics_to_remove)
    
    X_k_filtered = X_k;
    
    if strcmp(filter_type, 'notch')
        for i = 1:length(harmonics_to_remove)
            h = harmonics_to_remove(i);
            
            % Positive frequency component
            idx_pos = h + 1;
            % Negative frequency (conjugate)
            idx_neg = N - h + 1;
            
            % Zero out both components
            X_k_filtered(idx_pos) = 0;
            X_k_filtered(idx_neg) = 0;
        end
    end
    
    % Calculate THD before/after
    filter_info = calculate_filter_performance(X_k, X_k_filtered);
end
    \end{lstlisting}
\end{frame}

\begin{frame}{Harmonic Filtering - Concept}
    \begin{block}{Frequency Domain Filtering}
        Manipulate DTFS coefficients to remove unwanted harmonics.
    \end{block}
    
    \vspace{0.3cm}
    
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Filter Types Implemented:}
            \begin{enumerate}
                \item \textbf{Notch Filter:} Remove specific harmonics
                \item \textbf{Bandpass Filter:} Retain harmonics in range
                \item \textbf{Lowpass Filter:} Keep fundamental + few harmonics
            \end{enumerate}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Process:}
            \begin{enumerate}
                \item Perform DTFS analysis
                \item Zero out unwanted harmonics
                \item Perform inverse DTFS
                \item Get filtered time-domain signal
            \end{enumerate}
        \end{column}
    \end{columns}
    
    \vspace{0.3cm}
    
    \begin{alertblock}{Example}
        For motor drives: Remove 5th and 7th harmonics using notch filter.
    \end{alertblock}
\end{frame}

                \item Power factor: 0.980

        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Mitigation:}
            \begin{itemize}
                \item Passive 3rd harmonic filters
                \item Delta-wye transformers
                \item High-quality LED drivers
                \item K-4 rated transformers
            \end{itemize}
            
            \vspace{0.3cm}
            
            \textbf{Standards:}
            \begin{itemize}
                \item IEEE 519-2014: Non-compliant
                \item IEC 61000-3-2: Check limits
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Scenario Analysis: Motor Drive (VFD)}
    \begin{block}{Characteristics}
        Worst power quality with THD of 26.7\%. 5th and 7th harmonics at 20\% and 14\%.
    \end{block}
    
    \vspace{0.3cm}
    
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Causes:}
            \begin{itemize}
                \item 6-pulse rectifiers
                \item Follows $h = 6k \pm 1$
                \item Non-linear DC bus charging
            \end{itemize}
            
            \vspace{0.3cm}
            
            \textbf{Impact:}
            \begin{itemize}
                \item 3.5× above IEEE limit
                \item High K-factor: 4.55
                \item Requires K-4 transformers
                \item Significant heating
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Mitigation Results:}
            \begin{itemize}
                \item Notch filter: 61.7\% THD reduction
                \item Final THD: 10.8\%
                \item Power factor: 0.964 → 0.9942
                \item K-factor: 4.55 → 2.59
            \end{itemize}
            
            \vspace{0.3cm}
            
            \textbf{Recommendations:}
            \begin{itemize}
                \item Upgrade to 12-pulse rectifiers
                \item Active harmonic filters
                \item Line reactors (5\% impedance)
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Scenario Analysis: Data Center}
    \begin{block}{Characteristics}
        Multiple odd harmonics with THD of 20.4\% from SMPS loads.
    \end{block}
    
    \vspace{0.3cm}
    
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Causes:}
            \begin{itemize}
                \item Switch-mode power supplies
                \item High-frequency switching
                \item Distributed harmonic content
            \end{itemize}
            
            \vspace{0.3cm}
            
            \textbf{Impact:}
            \begin{itemize}
                \item K-factor: 2.89
                \item Power factor: 0.980
                \item Uniform harmonic distribution
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Mitigation:}
            \begin{itemize}
                \item Active harmonic filters
                \item K-13 rated transformers
                \item Hybrid passive-active systems
                \item Distributed PF correction
            \end{itemize}
            
            \vspace{0.3cm}
            
            \textbf{Challenges:}
            \begin{itemize}
                \item Variable load conditions
                \item Multiple harmonic frequencies
                \item Passive filters less effective
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Filtering Effectiveness Analysis}
    \begin{block}{Frequency Domain Notch Filtering}
        Achieved excellent removal of targeted harmonics (100\% for 5th and 7th).
    \end{block}
    
    \vspace{0.3cm}
    
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Advantages:}
            \begin{itemize}
                \item Precise harmonic removal
                \item No phase distortion
                \item Preserves fundamental
                \item Predictable results
            \end{itemize}
            
            \vspace{0.3cm}
            
            \textbf{Limitations:}
            \begin{itemize}
                \item Residual THD (10.8\%)
                \item 11th \& 13th remain
                \item Time-domain only
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Practical Implementation:}
            \begin{itemize}
                \item Active filters required
                \item Proper impedance matching
                \item Continuous monitoring
                \item Avoid resonance
            \end{itemize}
            
            \vspace{0.3cm}
            
            \textbf{Future Improvements:}
            \begin{itemize}
                \item Remove all harmonics $>$ 3rd
                \item Adaptive filtering
                \item Real-time implementation
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\section{Conclusion}

\begin{frame}{Key Contributions}
    \begin{enumerate}
        \item \textbf{DTFS Implementation:} Direct implementation providing accurate harmonic extraction with $O(N^2)$ complexity. Analysis completes in 0.05 seconds for $N=200$.
        
        \vspace{0.2cm}
        
        \item \textbf{Scenario Characterization:} Distinct harmonic signatures identified for LED lighting (3rd dominant), motor drives (5th \& 7th), and data centers (distributed).
        
        \vspace{0.2cm}
        
        \item \textbf{IEEE Compliance:} All non-linear loads violate IEEE 519-2014 (THD $>$ 8\%), with motor drives worst at 26.7\%.
        
        \vspace{0.2cm}
        
        \item \textbf{Filtering Performance:} Frequency-domain notch filtering achieved 61.7\% THD reduction (26.7\% → 10.8\%) and power factor improvement (0.966 → 0.994).
        
        \vspace{0.2cm}
        
        \item \textbf{Practical Framework:} Actionable recommendations for harmonic mitigation, transformer K-factor selection, and IEEE compliance.
    \end{enumerate}
\end{frame}
